package com.company;


import java.util.Random;
import java.util.Scanner;

public class Main {

    public static void main(String[] args) {


    { //counts mean for x and y cordinates were the pixel is 0;
      //The array is filled randomly from 0-4
        Scanner s=new Scanner(System.in);
        System.out.println("Please submit 2 numbers: first for rows, second for colums");
        int m=s.nextInt();
        int n=s.nextInt();
        int[][] arr = new int[m][n];
        Random r = new Random();
        int sumofi = 0;
        int sumofj = 0;
        int count = 0;


        for (int i = 0; i < arr.length; i++) {
            for (int j = 0; j < arr[i].length; j++) {
                arr[i][j] = r.nextInt(4);
                if (arr[i][j] == 0) {
                    sumofi += i;
                    sumofj += j;
                    count++;
                }
            }
        }
        for (int i = 0; i < arr.length; i++) {
            for (int j = 0; j < arr[i].length; j++)
                System.out.print(arr[i][j] + " ");
            System.out.println();
        }
        System.out.println();
        double meanofi = sumofi  / count;
        double meanofj=sumofj/count;
        System.out.print("mean of x is " + meanofi + " and mean of y is " + meanofj);
    }
}}
